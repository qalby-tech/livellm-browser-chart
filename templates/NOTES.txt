  # Port-forward to access the services locally:
  
  # noVNC web interface (recommended):
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "livellm-browser.fullname" . }} 6901:{{ .Values.service.novnc.port }}
  # Then open: http://localhost:6901
  
  # FastAPI Controller:
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "livellm-browser.fullname" . }} 8000:{{ .Values.service.api.port }}
  # Then open: http://localhost:8000
  
  # VNC (for VNC client):
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "livellm-browser.fullname" . }} 5901:{{ .Values.service.vnc.port }}
  # Then connect with VNC client to: localhost:5901
